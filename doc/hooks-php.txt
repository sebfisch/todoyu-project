Event Hooks
***************************************

	@param	Integer		$idProject

	@param	Integer		$idProject

	@param	Integer		$idProject

#project.delete
	Project deleted
	@param	Integer		$idProject

#projectpresetdata
	Collect preset data attributes (labels and value)
	@todo	Refactor
	@param	Integer		$idProject

	@param	Integer		$idProject
	@return	Array		[class, title, content]

#task.create.render
	Before rendering the form to create a new task

#projects.render
	Before projects are rendered for listings

#task.update
